```python
# 备份自动化脚本
backup_automation = {
    "完整备份脚本": """
#!/bin/bash
# 完整备份脚本
DATE=$(date +%Y%m%d)
BACKUP_DIR="/backup/full"
PROJECT_DIR="/path/to/project"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 执行完整备份
tar -czf $BACKUP_DIR/project_full_$DATE.tar.gz $PROJECT_DIR

# 清理旧备份
find $BACKUP_DIR -name "*.tar.gz" -mtime +365 -delete
""",
    "增量备份脚本": """
#!/bin/bash
# 增量备份脚本
DATE=$(date +%Y%m%d)
BACKUP_DIR="/backup/incremental"
PROJECT_DIR="/path/to/project"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 执行增量备份
rsync -av --delete $PROJECT_DIR/ $BACKUP_DIR/incremental_$DATE/

# 清理旧备份
find $BACKUP_DIR -type d -mtime +30 -exec rm -rf {} \;
"""
}
```
